npm init -y ----->>> برای ساخت پکیج جیسون و تنظیمات نود جی اس
---------------------------------------------------------------------------
npm install express ----->>> برای نصب فریم ورک اکسپرس برای ساخت ای پی آی
---------------------------------------------------------------------------
برای ایمپورت کردن اکسپرس داخل پروژه
------>>>{
const express = require('express');
const app = express();
}
---------------------------------------------------------------------------
برای ساخت یک روت و نمایش یه پیام در صفحه
------->>>{
app.get('/', (req, res) => {
  res.send('Node.js');
});
}
---------------------------------------------------------------------------
اجرای سرور روی پورت مشخص
------->>>{
app.listen(3000, () => {
  console.log('سرور اجرا شد');
});
}
--------------------------------------------------------------------------
